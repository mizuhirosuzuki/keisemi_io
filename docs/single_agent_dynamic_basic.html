            [:mileage_lag, :mileage, :action_lag] =>

                (mileage_lag, mileage, action_lag) ->
                (
                    (
                        (action_lag == 0) & 
                        (5 <= mileage_lag <= 95) & 
                        (mileage_lag == mileage)
                        ) |
                    ((action_lag == 1) & (mileage == 0))
                    ) ? "cond_obs_mileage1" :
                (
                    (
                        (action_lag == 0) & 
                        (5 <= mileage_lag <= 90) & 
                        (mileage_lag == mileage - 5)
                        ) |

                        (action_lag == 0) & 
                        (5 <= mileage_lag <= 90) &

            :cond_obs_mileage
