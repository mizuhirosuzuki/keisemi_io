Pi gen 
function pi1gen(theta)

    base = [
        0,
        0,
        theta[1] + theta[3],
        theta[1] + theta[2] + theta[3],
        0,
        0,
        theta[1],
        theta[1] + theta[2]
    ];

    invdiv = [theta[4], 0, theta[5]];

    base = reshape(repeat(base, 3), (8, :));
    invdiv = reshape(repeat(invdiv, 8), (:, 8))';
    return base + invdiv;
end

function pi2gen(theta)

    base = [
        0,
        theta[6] + theta[8],
        0,
        theta[6] + theta[7] + theta[8],
        0,
        theta[6],
        0,
        theta[6] + theta[7]
    ];

    invdiv = [theta[9], 0, theta[10]];

    base = reshape(repeat(base, 3), (8, :));
    invdiv = reshape(repeat(invdiv, 8), (:, 8))';
    return base + invdiv;
end
